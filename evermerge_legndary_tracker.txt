            <h2 class="text-xl font-semibold mt-10 mb-4 border-b border-gray-700 pb-2 opacity-50">Klart! ✅</h2>

            <h3 id="modal-title" class="text-xl font-bold mb-4 text-center"></h3>
            <div class="grid grid-cols-2 gap-4" id="tier-inputs">
                <!-- Inputs genereras här -->
            </div>
            <div class="mt-6 flex gap-3">
                <button onclick="closeModal()" class="flex-1 py-3 bg-gray-700 rounded-xl font-bold">Avbryt</button>
                <button onclick="saveData()" class="flex-1 py-3 bg-yellow-600 rounded-xl font-bold">Spara</button>

    <script>
        const animals = [
            "Phoenix", "Babe the blue ox", "Thunderbird", "Golden Ram", "Loch Ness Monster",
            "Dragon", "Pegasus", "T-Rex", "Bandersnatch", "King Bear", "Cockatrice",
            "Chinese dragon", "Kraken", "Jabberwock", "Hippogriff", "Basilisk", "Roc",
            "Chupacabra", "Enfield", "Jubjub Bird"
        ];

        // Vikter baserat på 3x T6 mål (Dvs hur många T1 varje tier är värd för att nå målet)
        // Mål = 1 final = 3st T6.
        // T6 = 1 enhet (behöver 3)
        // T5 = 1/2.66... nja, vi använder användarens siffror för totalt behov:
        const tierRequirements = [315, 125, 50, 20, 8, 3]; // Behov av respektive tier för att få 1 final (vid 3-merge T6)
        
        // För att räkna % behöver vi veta hur mycket värde varje djur i en tier tillför i % av slutmålet
        const tierWeights = [
            100/315, // T1
            100/125, // T2
            100/50,  // T3
            100/20,  // T4
            100/8,   // T5
            100/3    // T6
        ];

        let state = {};
        let currentEditing = null;

        function init() {
            const saved = localStorage.getItem('evermerge_tracker_v1');
            if (saved) {
                state = JSON.parse(saved);
            } else {
                animals.forEach(a => {
                    state[a] = { tiers: [0, 0, 0, 0, 0, 0], completed: false };
                });
            }
            render();
        }

        function calculateProgress(animalName) {
            const data = state[animalName];
            if (data.completed) return 100;
            
            let totalProgress = 0;
            data.tiers.forEach((count, index) => {
                totalProgress += count * tierWeights[index];
            });
            
            return Math.min(Math.round(totalProgress), 100);
        }

        function render() {
            const activeList = document.getElementById('active-list');
            const completedList = document.getElementById('completed-list');
            const podium = document.getElementById('podium');
            
            activeList.innerHTML = '';
            completedList.innerHTML = '';

            // Sortera djur efter progress
            const sortedAnimals = animals.map(name => ({
                name,
                progress: calculateProgress(name),
                completed: state[name].completed
            })).sort((a, b) => b.progress - a.progress);

            // Pallplats (Top 3 aktiva)
            const top3 = sortedAnimals.filter(a => !a.completed).slice(0, 3);
            podium.innerHTML = '';
            
            const podiumLabels = ['🏆 Guld', '🥈 Silver', '🥉 Brons'];
            const podiumClasses = ['gold text-gray-900', 'silver text-gray-900', 'bronze text-gray-900'];

            top3.forEach((animal, i) => {
                const card = document.createElement('div');
                card.className = `${podiumClasses[i]} p-3 rounded-xl text-center shadow-lg transform scale-100 active:scale-95 transition-all`;
                card.innerHTML = `
                    <div class="text-[10px] font-black uppercase mb-1">${podiumLabels[i]}</div>
                    <div class="text-xs font-bold leading-tight truncate">${animal.name}</div>
                    <div class="text-lg font-black">${animal.progress}%</div>
                `;
                card.onclick = () => openEdit(animal.name);
                podium.appendChild(card);
            });

            // Listor
            sortedAnimals.forEach(animal => {
                const container = animal.completed ? completedList : activeList;
                const card = document.createElement('div');
                card.className = `bg-gray-800 p-4 rounded-2xl border border-gray-700 active:bg-gray-750 cursor-pointer ${animal.completed ? 'completed' : ''}`;
                
                card.innerHTML = `
                    <div class="flex justify-between items-center mb-2">
                        <span class="font-bold text-lg">${animal.name}</span>
                        <span class="text-sm font-mono ${animal.progress >= 90 ? 'text-green-400' : 'text-gray-400'}">${animal.progress}%</span>
                    </div>
                    <div class="w-full bg-gray-700 h-3 rounded-full overflow-hidden">
                        <div class="h-full bg-yellow-500 transition-all duration-500" style="width: ${animal.progress}%"></div>
                    </div>
                    <div class="flex gap-2 mt-3 overflow-x-auto text-[10px] text-gray-400 no-scrollbar">
                        ${state[animal.name].tiers.map((count, i) => `
                            <span class="bg-gray-900 px-2 py-1 rounded">T${i+1}: <b class="text-gray-200">${count}</b></span>
                        `).join('')}
                    </div>
                    ${!animal.completed && animal.progress >= 100 ? `
                        <button onclick="markDone(event, '${animal.name}')" class="mt-3 w-full py-2 bg-green-700 rounded-lg text-xs font-bold">Markera som klar!</button>
                    ` : ''}
                `;
                
                card.onclick = () => openEdit(animal.name);
                container.appendChild(card);
            });

            localStorage.setItem('evermerge_tracker_v1', JSON.stringify(state));
        }

        function openEdit(name) {
            currentEditing = name;
            document.getElementById('modal-title').innerText = name;
            const container = document.getElementById('tier-inputs');
            container.innerHTML = '';
            
            state[name].tiers.forEach((val, i) => {
                const div = document.createElement('div');
                div.className = "flex flex-col";
                div.innerHTML = `
                    <label class="text-xs text-gray-400 mb-1">Tier ${i+1}</label>
                    <input type="number" id="t-input-${i}" value="${val}" 
                        class="bg-gray-700 p-3 rounded-xl text-center text-xl font-bold focus:ring-2 focus:ring-yellow-500 outline-none"
                        onclick="this.select()">
                `;
                container.appendChild(div);
            });

            if(state[name].completed) {
                const unfinishBtn = document.createElement('button');
                unfinishBtn.className = "col-span-2 mt-2 text-xs text-red-400 underline";
                unfinishBtn.innerText = "Ångra 'Klar'-status";
                unfinishBtn.onclick = () => { state[name].completed = false; closeModal(); render(); };
                container.appendChild(unfinishBtn);
            }

            document.getElementById('edit-modal').classList.remove('hidden');
        }

        function closeModal() {
            document.getElementById('edit-modal').classList.add('hidden');
        }

        function saveData() {
            const tiers = [];
            for(let i=0; i<6; i++) {
                tiers.push(parseInt(document.getElementById(`t-input-${i}`).value) || 0);
            }
            state[currentEditing].tiers = tiers;
            closeModal();
            render();
        }

        function markDone(e, name) {
            e.stopPropagation();
            state[name].completed = true;
            render();
        }

        window.onload = init;
    </script>
